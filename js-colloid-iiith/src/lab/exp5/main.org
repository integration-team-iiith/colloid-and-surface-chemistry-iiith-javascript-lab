	water: function(){
		if(this.step_no == 0)
		{
			$("#back1").show();
			$("#beak1").show();
			this.animateDirection_1('beak1', '+=38', '-=5', '1000');
			this.step_no++;
			setTimeout(function(){
			$("#flask1").show();
			$("#arrow1").show();
			$("#ar1_txt").show();
			},1000);
			setTimeout(function(){
				this.x=1;
				$("#arrow1").hide();
				$("#ar1_txt").hide();
				time1=setInterval(function(){
					images=[];
					for(i=1;i<=69;i++)
					{
						images[i]= "images/sprites/water_flasks/"  + i + ".png";
					}
					if(this.x<images.length)
					{
						$("#flask1").attr("src",images[this.x]);
					}
					this.x++;
					if(this.x ==70){
						$("#flask1").hide();
						clearInterval(time1);
					}
				},50);
			},3000);
#+END_SRC
